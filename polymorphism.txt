class Animal {
    String name;

    Animal(String name) {
        this.name = name;
    }

        void makeSound() {
        System.out.println("The " + name + " makes a generic animal sound.");
    }
}

class Dog extends Animal {
    Dog(String name) {
        super(name); 
	}

    void makeSound() {
        System.out.println("The " + name + " says: Woof! Woof!");
    }
}

class Cat extends Animal {
    Cat(String name) {
        super(name); // Call the Animal constructor
    }
    void makeSound() {
        System.out.println("The " + name + " says: Meow.");
    }
}

public class Zoo {
    public static void main(String[] args) {
      
        Animal animal1 = new Dog("German Shepherd");
        Animal animal2 = new Cat("Siamese");
        Animal animal3 = new Animal("Unspecified Creature");


        System.out.println("--- Zoo Animal Sounds ---");

        System.out.print("Animal 1: ");
        animal1.makeSound(); 
        
        System.out.print("Animal 2: ");
        animal2.makeSound(); 
        
        System.out.print("Animal 3: ");
        animal3.makeSound(); 
        

        System.out.println("\n--- Using a Polymorphic Array ---");

        Animal[] zoo = new Animal[3];
        zoo[0] = new Dog("Poodle");
        zoo[1] = new Cat("Persian");
        zoo[2] = new Dog("Labrador");

            for (Animal animal : zoo) {
            System.out.print("Array Element: ");
            animal.makeSound();
        }
    }
}


//OUTPUT
--- Zoo Animal Sounds ---
Animal 1: The German Shepherd says: Woof! Woof!
Animal 2: The Siamese says: Meow.
Animal 3: The Unspecified Creature makes a generic animal sound.

--- Using a Polymorphic Array ---
Array Element: The Poodle says: Woof! Woof!
Array Element: The Persian says: Meow.
Array Element: The Labrador says: Woof! Woof!
